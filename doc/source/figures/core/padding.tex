\def\f#1#2{int((#2)*12+(#1)+1)}

\def\trunc#1{(#1) >= 0 ? (((#1) < 12) ? (#1) : 11) : 0}
\def\wrap#1{Mod(#1, 12)}
\def\mirrorB#1{(#1) >= 12 ? (23 - (#1)) : (#1)}
\def\mirror#1{\mirrorB{Mod(#1,24)}}
\def\padding#1#2#3{\f{#1{#2}}{#1{#3}}}
\def\inR#1{(#1 >= 0 && #1 < 12)}
\def\fillA#1#2{(\inR{#1} && \inR{#2}) ? \f{#1}{#2} : 69}
